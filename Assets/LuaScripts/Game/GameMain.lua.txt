---注册挂起/恢复应用时回调
---@type fun(pause:boolean):void
---@param pause boolean @ 应用是否挂起
function GameApplicationPauseCallback(pause)
    if LoginManager:IsDelUser() then
        return
    end
    if pause == false then
        LoginManager:TryReLogin()
    end
end
