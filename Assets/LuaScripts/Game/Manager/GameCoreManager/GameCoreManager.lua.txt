---启动游戏检查结束队列
function GameCoreManager:StartCheckGameOverMsgQueue()
    self:StopCheckGameOverMsgQueue()
    self._HandleGameOverMsgQueueCoroutine = CoroutineHelper:StartAlone(function()
        self:CheckGameOverMsgQueueCoroutine()
    end)
end

---停止游戏检查结束队列
function GameCoreManager:StopCheckGameOverMsgQueue()
    if self._HandleGameOverMsgQueueCoroutine then
        CoroutineHelper:StopAlone(self._HandleGameOverMsgQueueCoroutine)
    end
    self._HandleGameOverMsgQueueCoroutine = nil
    self._HandleGameOverMsgQueue:Clear()
end

---检查游戏结束队列协程
---@type fun():void
function GameCoreManager:CheckGameOverMsgQueueCoroutine()
    --PrintInfo("游戏结束队列协程开始")
    local waitUtil1 = nil
    local waitUtil2 = nil
    ---@type HandleTriggerGameOverTypeData
    local triggerGameOverTypeData = nil
    while self._IsGameOver ~= true do
        --PrintInfo("游戏结束队列协程等待消息  111")
        ---此处是为了减少yield处理，会多等待一帧
        if self._HandleGameOverMsgQueue:Count() <= 0 then
            waitUtil1 = waitUtil1 or CS.UnityEngine.WaitUntil(function()
                return self._HandleGameOverMsgQueue:Count() > 0
            end)
            coroutine.yield(waitUtil1)
        end
        --PrintInfo("游戏结束队列协程等待消息  222")
        ---@type HandleTriggerGameOverTypeData
        triggerGameOverTypeData = self._HandleGameOverMsgQueue:Peek()
        if triggerGameOverTypeData then
            if self._IsGameOver ~= true then
                if self._CorePlayMgrData:IsTriggerLimit(triggerGameOverTypeData.gameOverType) then
                    self:FireTriggerGameOver(triggerGameOverTypeData.gameOverType, triggerGameOverTypeData.bounds)
                else
                    self:CheckGameResult(triggerGameOverTypeData.gameOverType, triggerGameOverTypeData.bounds)
                end
                waitUtil2 = waitUtil2 or CS.UnityEngine.WaitUntil(function()
                    return triggerGameOverTypeData.isHandle == true
                end)
                coroutine.yield(waitUtil2)
                --note:变量声明移出while之后，每次循环结束都要重置为nil
                triggerGameOverTypeData = nil
            end
        end
    end
    --PrintInfo("游戏结束队列协程结束")
end

---将关卡产出的分数提示插入到退出提示里
---@type fun():void
function GameCoreManager:SetupLevelChallengePointsExitTip()
    ActivityMgr:TryLevelExitTipRunningActivity()
    ---@param activityScore LevelChallengeScore
    for _, activityScore in pairs(self._CorePlayMgrData:GetActivityPoints()) do
        local activityCfg = EventDataMgr:GetEventDataById(activityScore.activityId)
        if activityCfg.ExitTipsID then
            ---不检查二次 ， 只检查三次
            --local exist = self:FindExitSureTip(activityCfg.ExitTipsID)
            --if exist then
            --    exist.num = activityScore.double
            --elseif Tables.ExitTips[activityCfg.ExitTipsID] then
            --    self:RegisterExitSureTip(activityCfg.ExitTipsID,activityScore.double)
            --end

            if activityScore.tipsAgain then

---使用id查询
---@type fun(id):table
---@param id : @ 提示的ID
---@return table
function GameCoreManager:FindExitSureTip(id)
    for _, tip in ipairs(self._ExitSureTip) do
        if tip.id == id then
            return tip
        end
    end
    return nil
end

---是否有退出二次确认提示
---@type fun():boolean
---@return boolean
function GameCoreManager:HasExitSureTip()
    return not TableIsEmpty(self._ExitSureTip)
end

---消费一个三次提示
---@type fun():LevelChallengeScore
---@return LevelChallengeScore @ 三度提示
function GameCoreManager:UseExitSureTipAgain()
    local k, v = next(self._ExitSureTipAgain)
    if k then
        self._ExitSureTipAgain[k] = nil
    end
    return v
end

---注册退出二次确认提示
---@type fun(id, num):void
---@param id string @ 提示的ID
---@param strNum string @ 提示的数字字符串
---@return void
function GameCoreManager:RegisterExitSureTip(id, strNum)
    local tip = self:FindExitSureTip(id)
    if tip then
        tip.strNum = strNum
    else
        table.insert(self._ExitSureTip, { id = id, strNum = strNum })
    end
end

---注销退出二次确认提示
---@type fun(id):void
---@param id string @ 提示的ID
---@return void
function GameCoreManager:UnRegisterExitSureTip(id)
    for i, tip in ipairs(self._ExitSureTip) do
        if tip.id == id then
            table.remove(self._ExitSureTip, i)
            break
        end
    end
end

---获取限制条件ID
---@type fun(limitType:EGameOverType):string
---@param limitType EGameOverType @限制类型
---@return string @限制条件ID
function GameCoreManager:GetLimitIdByLimitType(limitType)
    if limitType == EGameOverType.LostGuest then
        return EGoalId.NotLostGuest
    elseif limitType == EGameOverType.LostItem then
        return EGoalId.NotLostItem
    elseif limitType == EGameOverType.GuestAnger then
        return EGoalId.NotGuestAnger
    elseif limitType == EGameOverType.BurnedFood then
        return EGoalId.NotBurnedFood
    end
    return nil
end
