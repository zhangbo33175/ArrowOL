---初始化
---@type fun():void

    ChapterListManager.super.Init(self)

    HonorEventHelper:AddListener(EventCmd.TimerAreaOfflineCoinCDOver, handler(self, self.OnTimerAreaOfflineCoinCDOverCallback))
    HonorEventHelper:AddListener(EventCmd.OnLevelSuccess, handler(self, self.OnLevelSuccess))
    -- 启动时恢复存档中所有区域的CD时间事件
    --for areaId, timeStamp in pairs(PbRootData.GameData.cdEnds) do
    --    self:RegisterAreaOfflineCD(areaId, timeStamp)
    --end
    -- 启动时检查是否有更新的章节需要自动解锁
    local checkNextChapterId = tostring(tonumber(self:GetLatestChapterId()) + 1)
    if ChapterDataManager:GetChapterDataByChapterId(checkNextChapterId) ~= nil then
        local meetCondition = self:MeetConditionForUnlock(checkNextChapterId)
        -- note:特殊处理，检查是否存在待领取的奖励，没有才直接跨入下一章
        if meetCondition then
            if not self:CheckIsPauseUnlockNextChapter(self:GetLatestChapterId()) then
                -- 直接解锁下一章，没有提示过程
                self:UnlockChapter(checkNextChapterId)
            else
                self:OpenChapterUnlockUI(checkNextChapterId)
            end

        end
    end
    -- 启动时默认设置最新的区域Id和章节Id为运行时区域Id和章节Id
    self:SetRuntimeChapter(self:GetLatestChapterId())
    --设置最大章节ID( 要调用在 CheckIsPauseUnlockNextChapter 之后 )
    self:SetupMaxChapterID()
end

---是否打成解锁指定章节的条件（比较lv）
---@type fun(chapterId :string):boolean
---@param chapterId string @ 章节ID
function ChapterListManager:MeetConditionForUnlock(chapterId)
    local lastChapterData = ChapterDataManager:GetChapterDataByChapterId(tostring(tonumber(chapterId) - 1))
    local lastChapterLv = self:GetRuntimeChapterLvByChapterId(lastChapterData.ID) - 1
    local lastChapterMaxLv = lastChapterData.UnlockNextLevel - 1
    return lastChapterLv >= lastChapterMaxLv
end

---是否为已解锁章节

---@param chapterId string @章节Id

function ChapterListManager:IsUnlockedChapter(chapterId)

---是否已完成指定章节
---@type fun(chapterId:string):boolean
---@param chapterId string @章节Id
---@return boolean 是否已完成
function ChapterListManager:IsChapterCompleted(chapterId)
    local curLv = self:GetRuntimeChapterLvByChapterId(chapterId)
    local unlockNextLv = ChapterListManager:GetChapterDataByChapterId(chapterId).UnlockNextLevel
    return curLv >= unlockNextLv
end

---是否已完成指定区域的所有章节
---@type fun(areaId:string):boolean
---@param areaId string @区域Id
---@return boolean 是否已完成
function ChapterListManager:IsAllChaptersCompleted(areaId)
    local chapterIDs = ChapterListManager:GetAreaDataByAreaId(areaId).ChaptersID
    for _, chapterId in pairs(chapterIDs) do
        if self:IsChapterCompleted(chapterId) == false then
            return false
        end
    end
    return true
end

---获取运行时章节进度指标
---@type fun(chapterId:string, progressIndex:number):number
---@param chapterId string @章节Id
---@param progressIndex number @进度指标的索引
---@return number 进度值
function ChapterListManager:GetRuntimeChapterProgressByChapterId(chapterId, progressIndex)
    local progress = 0
    if PbRootData.GameData.chapterInfos[chapterId] ~= nil then
        if PbRootData.GameData.chapterInfos[chapterId].progress ~= nil then
            progress = PbRootData.GameData.chapterInfos[chapterId].progress[progressIndex] == nil and 0 or PbRootData.GameData.chapterInfos[chapterId].progress[progressIndex]
        end
    end
    local result = progress * ChapterDataManager:GetChapterInfoDataByChapterInfoId(ChapterDataManager:GetChapterDataByChapterId(chapterId).ChapterInfoID[progressIndex]).Coef
    local decimalPointNum = #tostring(result % 1) - 2
    if decimalPointNum > 0 then
        result = tonumber(string.format(mstr { "%.<<decimalPointNum>>f", decimalPointNum = decimalPointNum }, result))
    end
    return result
end

---检查最新区域离线奖励推送状态并处理相应推送
---@type fun(areaId:string):void
---@param areaId string @ 最新区域Id
function ChapterListManager:UpdateChapterRewardNotification(areaId)
    local utc0TimeStamp = self:GetAreaOfflineRewardCoinsTimeStamp(areaId) or 0
    local timeValid = utc0TimeStamp >= HonorTimerHelper:GetUTC0TimeStamp()
    ---时间有效则更新，时间无效且存在有效推送则删除
    if timeValid then
        local localTimeStamp = HonorTimerHelper:GetLocalTimeStamp(utc0TimeStamp)
        RNotificationHelper:GenerateSingleNotification(Tables.Push['5'], ENotificationConstKey.ChapterReward, true, localTimeStamp, nil, nil)
    elseif not timeValid then
        RNotificationHelper:TryCancelNotification(ENotificationConstKey.ChapterReward, true)
    end
end

--region ChapterListSave

---设置章节Id
---@type fun(chapterId:string):void
---@param chapterId string @章节Id
function ChapterListManager:SetLatestChapterId(chapterId)
    PbRootData.GameData.latestChapterId = chapterId
    PbIO.GameData.Save_latestChapterId()

    -- 更新关卡游玩提醒
    self:UpdateLevelNotification(chapterId, false, true)
end

---@type fun(chapterId:string,isInit:boolean,isUnlock:boolean):void
---@param chapterId string @ 最新的章节ID
---@param isInit boolean @ 初始化时需要检查之前是否存在
---@param isUnlock boolean @ 是否在解锁新的章节
function ChapterListManager:UpdateLevelNotification(chapterId, isInit, isUnlock)
    ---给一个默认值
    chapterId = chapterId or self:GetLatestChapterId()

    local chapterConfig = ChapterDataManager:GetChapterDataByChapterId(chapterId)
    local extraVaild = RNotificationHelper:SearchVaildNotification(ENotificationConstKey.LevelPlay)
    ---初始化时存在一个之前的推送数据，则不必更新了
    if extraVaild and isInit then
        return
    end

    local name = GetLang(chapterConfig.ChapterTitle)
    ---@param t NotificationStrReplace
    RNotificationHelper:GeneratePeriodicNotification(Tables.Push['3'], ENotificationConstKey.LevelPlay, 0, isUnlock, nil, ArrayTableConvert(Tables.Push['3'].ContentReplace, function(t)
        return { key = t.key, value = name }
    end))
end

---@type fun():string

---获取章节Id
---@type fun():string
---@return string @章节Id
function ChapterListManager:GetLatestChapterId()
    local chapterId = PbRootData.GameData.latestChapterId
    if string.isNullOrEmpty(chapterId) then
        PbRootData.GameData.latestChapterId = "1"
        PbIO.GameData.Save_latestChapterId()
        return PbRootData.GameData.latestChapterId
    end
    return chapterId
end

---目标章节是否大于最新章节
---@type fun(targetChapterLv:ChapterLevel,latestChapterLv:ChapterLevel):boolean
---@param targetChapterLv ChapterLevel @ 目标关卡id
---@param latestChapterLv ChapterLevel @ 最新章节、关卡 （可空）
function ChapterListManager:LargeThanLatestChapterLv(targetChapterLv, latestChapterLv)
    latestChapterLv = latestChapterLv or GamerDataManager:GetLatestChapterLevel()
    local lessOrEqual = targetChapterLv.Chapter < latestChapterLv.Chapter or (targetChapterLv.Chapter == latestChapterLv.Chapter and targetChapterLv.Level <= latestChapterLv.Level)
    return not lessOrEqual
end

---目标章节是否小于最新章节
---@type fun(targetChapterLv:ChapterLevel,latestChapterLv:ChapterLevel):boolean
---@param targetChapterLv ChapterLevel @ 目标关卡id
---@param latestChapterLv ChapterLevel @ 最新章节、关卡 （可空）
---@return boolean @ 目标章节是否小于最新章节
function ChapterListManager:LessThanChapterLv(targetChapterLv, latestChapterLv)
    latestChapterLv = latestChapterLv or GamerDataManager:GetLatestChapterLevel()
    return targetChapterLv.Chapter < latestChapterLv.Chapter or (targetChapterLv.Chapter == latestChapterLv.Chapter and targetChapterLv.Level < latestChapterLv.Level)
end
