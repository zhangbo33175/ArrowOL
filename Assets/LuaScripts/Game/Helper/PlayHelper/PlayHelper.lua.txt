---构造函数

function PlayHelper:ctor()

    self._mainMinChapterId = 1                                          ---@type number                                 ---主线最小章节Id
    self._mainMaxChapterId = 1000                                       ---@type number                                 ---主线最大章节Id
    self._goldenTicketPlayNameTextColorConfigData = nil                 ---@type TextColorConfigData                    ---金票玩家名字颜色配置数据

---创建函数
---@type fun():PlayHelper
---@return PlayHelper
function PlayHelper:Create()
    local obj = PlayHelper.new()
    return obj
end

---是否是主线章节
---@type fun(chapterId:string):boolean
---@param chapterId string @章节Id
---@return boolean
function PlayHelper:IsMainChapter(chapterId)
    if string.isNullOrEmpty(chapterId) then
        return false
    end
    local chapterIdNum = tonumber(chapterId)
    return chapterIdNum >= self._mainMinChapterId and chapterIdNum <= self._mainMaxChapterId
end

---是否在游戏中
---@type fun():boolean
---@return boolean
function PlayHelper:IsInLevelState()
    if GameStateMachine:CurrentStateType() == GameStateCmd.PlayingState then
        return true
    end

    if GameStateMachine:CurrentStateType() == GameStateCmd.PlayingFoodTruckState then
        return true
    end

    return false
end

---是否在满足玩第一关的条件
---@type fun():boolean
---@return boolean
function PlayHelper:IsPlayFirstLevel()
    local chapterId = ChapterListMgr:GetLatestChapterId()
    if chapterId == "1" and GamerDataManager:GetLevelLvByChapterId(chapterId) == 1 then
        return true
    end
    return false
end

---是否在满足玩准备关卡的条件打
---@type fun(chapterId:string,levelId:number):boolean
---@param chapterId string @章节Id
---@param levelId number @关卡Id
---@return boolean
function PlayHelper:IsPlayShowPrepareLevel(chapterId, levelId)
    local data = Tables.Constants.ShowPrepareLevel.Value
    local chapterIdNum = tonumber(chapterId)
    if chapterIdNum > data.Chapter then
        return true
    end
    if chapterIdNum == data.Chapter and levelId >= data.Level then
        return true
    end
    return false
end

---开最新章节物品升级界面
---@type fun():void
---@return void
function PlayHelper:OpenLastLevelUpgradeUI()
    local curChapterId = GamerDataManager:GetCurChapterId()
    local latest = ChapterListMgr:GetLatestChapterId()
    if latest ~= curChapterId then
        local isReady = IncreaserHotfixHelper:IsReadyGroupRes(latest)
        if isReady then
            self:LoadMainMap(latest, function()
                self:OpenLevelUpgradeUI(latest)
            end)
        else
            -- 下载完成后 自动打开下载更新界面
            IncreaserHotfixHelper:SetNeedAutoOpenDownloadUpdateUI(true)
            IncreaserHotfixHelper:OpenUIIncreaserUpdate()
        end
    else
        self:OpenLevelUpgradeUI(curChapterId)
    end
end

---打开最后一关准备关卡面板
---@type fun(openFrom:EOpenUILevelPrepareFrom):void
---@param openFrom EOpenUILevelPrepareFrom @打开来源,默认为 EOpenUILevelPrepareFrom.Other
function PlayHelper:OpenLastLevelPrepareUI(openFrom)
    if openFrom == nil then
        openFrom = EOpenUILevelPrepareFrom.Other
    end
    local curChapterId = GamerDataManager:GetCurChapterId()
    local curLevelId = GamerDataManager:GetLevelLvByChapterId(curChapterId, true)
    if not self:IsPlayShowPrepareLevel(curChapterId, curLevelId) then
        SoundMgr:PlaySound(Tables.Sounds.LevelStart_Ding.Name)
        local levelData = self:LoadLevelData(curChapterId, curLevelId)
        self:CommonOnPlayGame(levelData)
    else
        local latest = ChapterListMgr:GetLatestChapterId()
        if latest ~= curChapterId then
            local isReady = IncreaserHotfixHelper:IsReadyGroupRes(latest)
            if isReady then
                self:LoadMainMap(latest, function()
                    self:OpenLevelPrepareUI(latest, GamerDataMgr:GetLevelLvByChapterId(latest, true), openFrom)
                end)
            else
                -- 下载完成后 自动打开下载更新界面
                IncreaserHotfixHelper:SetNeedAutoOpenDownloadUpdateUI(true)
                IncreaserHotfixHelper:OpenUIIncreaserUpdate()
            end
        else
            self:OpenLevelPrepareUI(curChapterId, curLevelId, openFrom)
        end
    end
end

---获取关卡数据
---@type fun(characterId:string,levelId:number):Tables.LevelData_Item
---@param chapterId string @章节Id
---@param levelId number @关卡Id
---@return Tables.LevelData_Item @关卡数据
function PlayHelper:LoadLevelData(chapterId, levelId)
    ---@type Tables.Chapters_Item
    local chapterConfig = Tables.Chapters[tostring(chapterId)]
    if chapterConfig and chapterConfig.MainLevels and not ChallengeLevelHelper:IsChallengeLevel(levelId) and levelId > chapterConfig.MainLevels[2] then
        --return self._loopLevelHelper:GetLoopLevelConfig(chapterConfig, levelId)
        return nil
    end

    local levelName = table.concat({ "LevelData_", chapterId, "_", levelId })
    local levelData = Tables[levelName]
    return levelData
end

---查询关卡倍率

---@return number @ 关卡倍率奖励
function PlayHelper:SearchLevelHardLevelReward(diff)
    if Tables.Constants.HardLevelReward and Tables.Constants.HardLevelReward.Value[diff] then
        return Tables.Constants.HardLevelReward.Value[diff]
    end

end

---开始游戏
---@type fun(levelData:Tables.LevelData_Item):boolean
---@param levelData Tables.LevelData_Item @关卡数据
---@return boolean @是否成功
function PlayHelper:PlayLevel(levelData)
    if levelData == nil then
        return false
    end
    if GameCoreManager:IsLifeEnough() and self:PlayLevelImmediate(levelData) then
        GameCoreManager:SubLife()
        return true
    end
    return false
end

---立即开始游戏
---@type fun(levelData:Tables.LevelData_Item):boolean
---@param levelData Tables.LevelData_Item @关卡数据
---@return boolean @是否成功
function PlayHelper:PlayLevelImmediate(levelData)
    if levelData then
        GameCoreManager:ClearData()
        GameCoreManager._CurLevelData = levelData
        GameCoreManager._CorePlayMgrData._UseDoublePoints = BuffMgr:IsBuffValid(EResourceIDs.DoublePoints) -- 进入关卡初始化是否使用双倍积分
        --self._loopLevelHelper:OnGameExecute(levelData)
        return true
    end
    PrintError("PlayLevelImmediate is not implement,LevelData is nil")
    return false
end

---公用的游戏开始界面
---@type fun(lvData:Tables.LevelData_Item,view:GameViewSuper,boostersLua:UIBoosterButton[]):boolean
---@param lvData Tables.LevelData_Item @ 关卡数据
---@param view GameViewSuper @界面 (可以为空)
---@param boostersLua UIBoosterButton[] @ 前置道具使用情况 (可以为空)
---@return boolean @是否成功
function PlayHelper:CommonOnPlayGame(lvData, view, boostersLua)
    ---为打点收集战前选中要使用的道具
    local preboosterUse = {}
    if boostersLua then
        for i = 1, #boostersLua do
            if boostersLua[i]:GetSelectedState() then
                table.insert(preboosterUse, boostersLua[i]:GetPropID())
            end
        end
    end

    if self:PlayLevel(lvData) then
        if boostersLua then
            for i = 1, #boostersLua do
                boostersLua[i]:OnConfirmBooster()
            end
        end

        ---塞打点数据
        local lv = GamerDataMgr:GetLevelLvByChapterId(lvData.ChapterId)
        if lvData.Difficulty == ELevelDifficultyType.Challenge then
            ThinkingAnalyticsHelper:OnLevelStart({ Chapter = lvData.ChapterId, Level = lv }, lvData.LevelId, lvData.Difficulty, TrackProperty.LevelStartResult.success, preboosterUse, 1)
        else
            ThinkingAnalyticsHelper:OnLevelStart({ Chapter = lvData.ChapterId, Level = lv }, lvData.LevelId, lvData.Difficulty, TrackProperty.LevelStartResult.success, preboosterUse, CollectionMgr:CheckHasPackOrCouldGainPack(lvData) and 1 or 0)
        end

        ---开始关卡
        if self:IsInLevelState() then
            -- 如果本身就在游戏中，则重玩
            PlayHudCycle:OnAction(EPlayHudCycleAction.RePlayGame)
        else
            GameStateMachine:ChangeState(GameStateCmd.PlayingState, nil)
        end
        if view then
            -- 关闭界面
            view:Close()
        end

        ADMgr:TryLoadInterAd()
        return true
    else
        ---塞打点数据
        local lv = GamerDataMgr:GetLevelLvByChapterId(lvData.ChapterId)
        if lvData.Difficulty == ELevelDifficultyType.Challenge then
            ThinkingAnalyticsHelper:OnLevelStart({ Chapter = lvData.ChapterId, Level = lv }, lvData.LevelId, lvData.Difficulty, TrackProperty.LevelStartResult.failed, preboosterUse, 1)
        else
            ThinkingAnalyticsHelper:OnLevelStart({ Chapter = lvData.ChapterId, Level = lv }, lvData.LevelId, lvData.Difficulty, TrackProperty.LevelStartResult.failed, preboosterUse, CollectionMgr:CheckHasPackOrCouldGainPack(lvData) and 1 or 0)
        end

        --弹出体力面板
        if not GameCoreManager:IsLifeEnough() then
            OpenUISync(UIs.UILife)
        else
            if view then
                -- 关闭界面
                view:Close()
            end
        end
        return false
    end
end

---显示食物Item
---@type fun(foodId:string,parentTransform:UnityEngine.Transform,offsetType : EFoodTransformOffset , orientation:EOrderMachineOrientation):FoodItem
---@param foodId string @食物Id
---@param parentTransform UnityEngine.Transform @父节点
---@param offsetType EFoodTransformOffset @ 食物的整体偏移配置类型
---@param orientation EOrderMachineOrientation @ 食物的方向(默认朝向左边)
---@return FoodItem @食物Item
function PlayHelper:ShowFoodItem(foodId, parentTransform, offsetType, orientation)
    if orientation == nil then
        orientation = EOrderMachineOrientation.Left
    end
    local foodItemGo = PoolMgr:GetWithHandle(
            RES.GameObject_FoodItem.Name,
            parentTransform,
            handler(self, function(_, foodItemGo)
                if foodItemGo == nil then
                    PrintError("ShowFoodItem handler foodItemGo is nil foodId:" .. foodId)
                    return
                end
                foodItemGo.transform.localPosition = CS.UnityEngine.Vector3.zero
                foodItemGo.transform.localRotation = CS.UnityEngine.Quaternion.identity
                local scale = CS.UnityEngine.Vector3.one
                if orientation == EOrderMachineOrientation.Right then
                    scale.x = -1 --食物超右 需要反转一下
                end
                foodItemGo.transform.localScale = scale
            end))
    if foodItemGo == nil then
        PrintError("ShowFoodItem foodItemGo is nil foodId:" .. foodId)
        return nil
    end
    ---@type FoodItem
    local foodItem = foodItemGo:GetComponent("LuaBehaviour").luaClass
    if foodItem == nil then
        PrintError("ShowFoodItem foodItem is nil foodId:" .. foodId)
        return nil
    end
    Try(function()
        foodItem:ShowFoodById(foodId, offsetType)
    end)
    return foodItem
end

---显示弹出气泡
---@type fun(text:string,parentTransform:UnityEngine.Transform,pos:UnityEngine.Vector3, heightOffset:number, sortOrder:number, posType:EPos, PopBubblePrefabName:string,lineNum:number,fontSize:number,autoWidth:boolean,popBubbleOrientation:EPopBubbleOrientation):PopBubbleUINode
---@param text string @文本
---@param parentTransform UnityEngine.Transform @父节点
---@param pos UnityEngine.Vector3 @位置
---@param heightOffset number @高度偏移
---@param sortOrder number @排序
---@param posType EPos @位置类型
---@param PopBubblePrefabName string @弹出气泡预制体名称,默认为 "RES.GameObject_PopBubbleUINode.Name"
---@param lineNum number @行数,默认2行
---@param fontSize number @字体大小,默认为 26
---@param autoWidth boolean @是否自动宽度,默认为 true
---@param popBubbleOrientation EPopBubbleOrientation @气泡方向,默认为 EPopBubbleOrientation.Down
---@return PopBubbleUINode @弹出气泡
function PlayHelper:ShowPopBubbleUINode(text, parentTransform, pos, heightOffset, sortOrder, posType, PopBubblePrefabName, lineNum, fontSize, autoWidth, popBubbleOrientation)
    if string.isNullOrEmpty(PopBubblePrefabName) then
        PopBubblePrefabName = RES.GameObject_PopBubbleUINode.Name
    end
    if posType == nil then
        posType = EPos.Local
    end
    if posType == EPos.World then
        pos = parentTransform:InverseTransformPoint(pos) -- 转换为本地坐标
    end
    local popNodeGo = PoolMgr:GetWithHandle(
            PopBubblePrefabName,
            parentTransform,
            handler(self, function(_, _PopNode)
                ---@type UnityEngine.RectTransform
                local rectTransform = _PopNode:GetComponent("RectTransform")
                if rectTransform then
                    pos.y = pos.y + heightOffset
                    -- rectTransform.anchoredPosition3D = pos
                    rectTransform.localPosition = pos
                end
                _PopNode.transform.localScale = CS.UnityEngine.Vector3.one
                _PopNode.transform.localRotation = CS.UnityEngine.Quaternion.identity
            end))
    ---@type PopBubbleUINode
    local popNodeLua = popNodeGo:GetComponent("LuaBehaviour").luaClass
    popNodeLua:Init(text, pos, heightOffset, sortOrder, lineNum, fontSize, autoWidth, popBubbleOrientation)
    return popNodeLua
end

                    pos.y = pos.y + heightOffset

                    rectTransform.localPosition = pos

                _PopNode.transform.localRotation = CS.UnityEngine.Quaternion.identity
            end))

---弹出奖励
---@type fun(rewardData:RewardData,parentTransform:UnityEngine.Transform, pos:UnityEngine.Vector3, sortOrder:number,delayTime:number):PopClaimRewardUINode,number
---@param rewardData RewardData @奖励数据 RewardData
---@param parentTransform UnityEngine.Transform @父节点
---@param pos UnityEngine.Vector3 @位置(UI世界坐标)
---@param sortOrder number @UI排序
---@param delayTime number @延迟时间
---@param startFlyCallback function @开始飞行回调
---@param prefabName string @弹出奖励预制体名称,默认为 "RES.GameObject_PopClaimRewardUINode.Name"
---@return PopClaimRewardUINode,number @弹出奖励,动画完成需要的时间(含有延迟时间)
function PlayHelper:PopClaimRewardUINode(rewardData, parentTransform, pos, sortOrder, delayTime, startFlyCallback, prefabName, showX)
    if string.isNullOrEmpty(prefabName) then
        prefabName = RES.GameObject_PopClaimRewardUINode.Name
    end
    local popNodeGo = PoolMgr:GetWithHandle(
            prefabName,
            parentTransform,
            handler(self, function(_, _PopNode)
                ---@type UnityEngine.RectTransform
                local rectTransform = _PopNode:GetComponent("RectTransform")
                if rectTransform then
                    local localPos = parentTransform:InverseTransformPoint(pos)
                    rectTransform.anchoredPosition3D = localPos
                end
                _PopNode.transform.localScale = CS.UnityEngine.Vector3.one
                _PopNode.transform.localRotation = CS.UnityEngine.Quaternion.identity
            end))
    ---@type PopClaimRewardUINode
    local popNodeLua = popNodeGo:GetComponent("LuaBehaviour").luaClass
    popNodeLua:Init(rewardData, sortOrder, showX)
    DelayFunction(delayTime, true, function()
        if startFlyCallback then
            startFlyCallback()
        end
        popNodeLua:PlayFlyAnim(function()
            -- 准备回收
            PoolMgr:Put(popNodeGo)
        end)
    end)
    return popNodeLua, popNodeLua:GetFlyAnimLength() + delayTime
end

---弹出多个奖励
---@type fun(rewardList:List<RewardData>,parentTransform:UnityEngine.Transform, pos:UnityEngine.Vector3,offPosX:number,intervalTime:number, sortOrder:number):number
---@param rewardList List @奖励数据 List<RewardData>
---@param parentTransform UnityEngine.Transform @父节点
---@param pos UnityEngine.Vector3 @位置(UI世界坐标)
---@param offPosX number @X轴偏移
---@param intervalTime number @间隔时间
---@param sortOrder number @UI排序
---@param startFlyCallback function @开始飞行回调
---@param prefabName string @弹出奖励预制体名称,默认为 "RES.GameObject_PopClaimRewardUINode.Name"
---@return number @动画完成需要的时间
function PlayHelper:PopClaimRewardsUINode(rewardList, parentTransform, pos, offPosX, intervalTime, sortOrder, startFlyCallback, prefabName, showX)
    local maxAnimTime = 0
    local localPos = parentTransform:InverseTransformPoint(pos) -- 转换为本地坐标
    ---@param i number
    ---@param v RewardData
    for i, v in ipairs(rewardList.items) do
        local newPos = Unity.Vector3.zero
        if i % 2 == 0 then
            newPos.x = localPos.x + offPosX * -1
        else
            newPos.x = localPos.x + offPosX
        end
        newPos.y = localPos.y
        newPos.z = localPos.z

        local worldPos = parentTransform:TransformPoint(newPos) -- 转换为世界坐标
        local _, animTime = self:PopClaimRewardUINode(v, parentTransform, worldPos, sortOrder, intervalTime * (i - 1), startFlyCallback, prefabName, showX)
        if animTime > maxAnimTime then
            maxAnimTime = animTime
        end
    end
    return maxAnimTime
end

---判断是否解锁与当前最新章节id与关卡id对比
---@type fun(chapterId:number,levelId:number):boolean
---@param chapterId number @章节Id 请注意，这个地方传入的是数字
---@param levelId number @关卡Id
function PlayHelper:IsUnlock(chapterId, levelId)
    local lastChapterId = ChapterListMgr:GetLatestChapterId()
    local lastChapterIdNum = tonumber(lastChapterId)
    if lastChapterIdNum > chapterId then
        return true
    end

    if lastChapterIdNum == chapterId then
        local lastLevelId = tonumber(GamerDataMgr:GetLevelLvByChapterId(lastChapterId))
        if lastLevelId >= levelId then
            return true
        end
    end

    return false
end

---播放游戏内BGM
---@type fun():void
function PlayHelper:PlayInGameBGM()
    local levelData = GameCoreManager._CurLevelData
    if levelData then
        if MapMgr.CurLoadMapType == MapType.FoodTruck then
            --- 快餐车模式
            FoodTruckPlayHelper:PlayInGameBGM(MapMgr.CurLoadChapterId)
            return
        end
        if MapMgr.CurLoadMapType == MapType.Main then
            --- 主线模式
            local peakId = levelData.PeakId
            local peakLevelData = PeakLevelsDataMgr:GetDataById(tostring(peakId))
            if peakLevelData and peakId ~= ELevelType.Normal and not string.isNullOrEmpty(peakLevelData.LevelBGM) then
                SoundMgr:PlayBGM(peakLevelData.LevelBGM)
            else
                self:PlayCommonInGameBGM()
            end
            return
        end
        if not string.isNullOrEmpty(MapMgr.CurLoadChapterId) then
            PrintError("PlayHelper:PlayInGameBGM() 未知的地图类型 chapterId" .. MapMgr.CurLoadChapterId)
        end
    end
end

---播放普通关卡背景音乐

        -- 挑战关卡，播放挑战关卡音乐
        SoundMgr:PlayBGM(Tables.Sounds.BGM_Challenge.Name)
        return
    end

    local chapterData = ChapterDataMgr:GetChapterDataByChapterId(MapMgr.CurLoadChapterId)
    if chapterData and chapterData.MapMusic and chapterData.MapMusic.ingame then
        SoundMgr:PlayBGM(chapterData.MapMusic.ingame)
    end
end

function PlayHelper:OpenGiftPackUIWaitForClose(rewardList, boxName, effectName, content, areaId, shouldCheckInLevelAndSubCoin, isShowBgBlurImg, playCloseAnimCallback)
    if shouldCheckInLevelAndSubCoin == nil then
        shouldCheckInLevelAndSubCoin = true
    end
    if isShowBgBlurImg == nil then
        isShowBgBlurImg = true
    end
    local isClose = false
    OpenUISync(UIs.UIGiftPack,
            { Rewards = rewardList,
              BoxName = boxName,
              TXEffect = effectName,
              Content = content,
              AreaId = areaId,
              CloseCallback = function()
                  isClose = true
              end,
              ShouldCheckInLevelAndSubCoin = shouldCheckInLevelAndSubCoin,
              IsShowBgBlurImg = isShowBgBlurImg,
              BeginPlayCloseAnimCallback = playCloseAnimCallback })
    while not isClose do
        coroutine.yield()
    end
end

---根据关卡难度获取倍率
---@type fun(levelDifficultyType:ELevelDifficultyType):number
---@param levelDifficultyType ELevelDifficultyType @关卡难度
---@return number @倍率
function PlayHelper:GetRateByLevelDifficulty(levelDifficultyType)
    local diffDatas = Tables.Constants.HardLevelEventReward.Value
    if levelDifficultyType == ELevelDifficultyType.Normal then
        return diffDatas[1]
    elseif levelDifficultyType == ELevelDifficultyType.Hard then
        return diffDatas[2]
    elseif levelDifficultyType == ELevelDifficultyType.SuperHard then
        return diffDatas[3]
    elseif levelDifficultyType == ELevelDifficultyType.Challenge then
        return diffDatas[4]
    else
        return diffDatas[1]
    end
end

---是否是有效的通行证金色头像框
---@type fun(frameId:string):boolean
---@param frameId string @头像框Id
---@return boolean @是否是通行证金色头像框
function PlayHelper:IsValidSeasonPassFrame(frameId)
    if string.isNullOrEmpty(frameId) then
        return false
    end

    if GoldenTicketActivityLogic:IsGoldenTicketFrame(frameId) and GoldenTicketActivityLogic:IsActivityInProgress() then
        return true
    end

    return false
end

---@param refuseAd boolean @ 切换章节时不要播放广告
function PlayHelper:LoadMainMap(chapterId, callback, refuseAd)

    MapMgr:SwitchMap(chapterId, callback, refuseAd)

---初始化道具图标
---@type fun(ui:LuaBehaviourSuper, iconImg:UnityEngine.UI.Image, propId:string, areaId:string):void
---@param ui LuaBehaviourSuper @LuaBehaviourSuper
---@param iconImg UnityEngine.UI.Image @图标
---@param propId string @道具Id
---@param areaId string @区域Id   
function PlayHelper:InitPropIcon(ui, iconImg, propId, areaId)
    if ui and iconImg and not string.isNullOrEmpty(propId) then
        local itemData = ResourceDataMgr:GetResourceData(propId)
        if itemData then
            if itemData.ID == EResourceIDs.Coin then
                self:SetCoinIconViewImage(ui, iconImg, areaId)
            else
                self:SetPropIconViewImageByItemData(ui, iconImg, itemData)
            end
        end
    end
end
