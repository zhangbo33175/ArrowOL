---检查错误码
---@type fun(returnData:table):boolean
---@param returnData table @传入的数据
---@return boolean @返回是否需要中断返回code
function SaveDataHelper:CheckErrorCode(returnData)
    if returnData ~= nil then
        if returnData["error_code"] == NetCode.Success[1] then
            return true
        elseif returnData["error_code"] == 1001 then
            -- 出现顶号了，必须下线,回调不执行
            PrintError("弹出下线操作")
            return false
        end
    end
    return true
end

---保存用户登陆信息
---@type fun(returnData:NetMsgDef.NetRGameSyncData.RGameLoginResp)
---@param returnData NetMsgDef.NetRGameSyncData.RGameLoginResp
function SaveDataHelper:SaveSignInInfos(returnData)
    if returnData then
        if returnData.user_id ~= nil and returnData.user_id > 0 then
            PbRootData.UserData.uid = returnData.user_id
            PbIO.UserData.Save_uid()
            ---UID存入
            SolarRoot.SDK.TGAHelper:ModifyDynamicCommonTrackProperty("uid", PbRootData.UserData.uid)
            SolarRoot.SDK.TGAHelper:Login(PbRootData.UserData.uid)
        end
        if returnData.create_time ~= nil and returnData.create_time > 0 then
            PbRootData.UserData.regTime.utc0 = returnData.create_time
            PbRootData.UserData.regTime.utc0String = TimerHelper:GetTimeStringByTimeStamp("yyyy-MM-dd HH:mm:ss", returnData.create_time)
            PbRootData.UserData.regTime.locality = TimerHelper:GetLocalTimeStamp(returnData.create_time)
            PbRootData.UserData.regTime.localString = TimerHelper:GetTimeStringByTimeStamp("yyyy-MM-dd HH:mm:ss", TimerHelper:GetLocalTimeStamp(returnData.create_time))
            PbIO.UserData.Save_regTime()
        end
        if returnData.login_time ~= nil and returnData.login_time > 0 then
            PbRootData.UserData.loginTime.last.utc0 = PbRootData.UserData.loginTime.current.utc0
            PbRootData.UserData.loginTime.last.utc0String = PbRootData.UserData.loginTime.current.utc0String
            PbRootData.UserData.loginTime.last.locality = PbRootData.UserData.loginTime.current.locality
            PbRootData.UserData.loginTime.last.localString = PbRootData.UserData.loginTime.current.localString
            PbRootData.UserData.loginTime.current.utc0 = returnData.login_time
            PbRootData.UserData.loginTime.current.utc0String = TimerHelper:GetTimeStringByTimeStamp("yyyy-MM-dd HH:mm:ss", returnData.login_time)
            PbRootData.UserData.loginTime.current.locality = TimerHelper:GetLocalTimeStamp(returnData.login_time)
            PbRootData.UserData.loginTime.current.localString = TimerHelper:GetTimeStringByTimeStamp("yyyy-MM-dd HH:mm:ss", TimerHelper:GetLocalTimeStamp(returnData.login_time))
            PbIO.UserData.Save_loginTime()
        end

        if returnData.binding_time ~= nil and returnData.binding_time > 0 then
            PbRootData.UserData.aidBindTime.utc0 = returnData.binding_time
            PbRootData.UserData.aidBindTime.utc0String = TimerHelper:GetTimeStringByTimeStamp("yyyy-MM-dd HH:mm:ss", returnData.binding_time)
            PbRootData.UserData.aidBindTime.locality = TimerHelper:GetLocalTimeStamp(returnData.binding_time)
            PbRootData.UserData.aidBindTime.localString = TimerHelper:GetTimeStringByTimeStamp("yyyy-MM-dd HH:mm:ss", TimerHelper:GetLocalTimeStamp(returnData.binding_time))
            PbIO.UserData.Save_aidBindTime()
        end

        if returnData.token ~= "" then
            -- 登陆成功后会有token
            self._loginToken = returnData.token;
        end
        if returnData.country ~= "" then
            -- 返回登陆后的国家
            UserTypeMgr:TrySaveUserCountry(returnData.country)
        end
        if returnData.ServerUserInfo ~= nil then
            -- 云端有存档,同步第三方账号
            PbRootData.GameData.aidData.aid = returnData.ServerUserInfo.server_account
            PbRootData.GameData.aidData.platform = returnData.channel_type
            PbIO.GameData.Save_aidData()
        end
    end
end

---设置服务器的信息到本地（更新uid aid gameData）
---@type fun(serverGameData:SavePBMsgDef.SaveGameData.GameData,uid:number,aid:string，eAidPlatform:EAidPlatform)
---@param serverGameData SavePBMsgDef.SaveGameData.GameData
---@param uid number @游戏服uid
---@param aid string @第三方账号
---@param eAidPlatform EAidPlatform @第三方平台
function SaveDataHelper:SaveServerUserInfoToLocalByGameData(serverGameData, uid, aid, eAidPlatform)
    if serverGameData then
        PbRootData.GameData = self:DynamicReplacementGameData(serverGameData)
        PbIO.GameData.SaveAll()
    end

    PbRootData.UserData.uid = uid
    PbIO.UserData.Save_uid()

    PbRootData.GameData.aidData.aid = aid
    PbRootData.GameData.aidData.platform = eAidPlatform
    PbIO.GameData.Save_aidData()
end

---动态替换存档数据
---@type fun(otherGameData:SavePBMsgDef.SaveGameData.GameData):table
---@param otherGameData SavePBMsgDef.SaveGameData.GameData @游戏数据
function SaveDataHelper:DynamicReplacementGameData(otherGameData)
    package.loaded["SaveGameData"] = nil
    require('SaveGameData')
    local result = PbRootData.GameData
    self:CopyGameDataTable(result, otherGameData)
    return result
end

---复制table,递归复制,originTab 会覆盖 desTab
---@type fun(desTab:table,originTab:table)
---@param desTab table @目标table
---@param originTab table @原始table
function SaveDataHelper:CopyGameDataTable(desTab, originTab)
    for key, oldValue in pairs(originTab) do
        if type(oldValue) ~= "table" then
            desTab[key] = oldValue
        else
            if desTab[key] == nil then
                desTab[key] = {}
            end
            self:CopyGameDataTable(desTab[key], oldValue)
        end
    end
end

---设置返回数据，登陆成功后可调用此方法
function SaveDataHelper:SaveLocalDataAndAutoSyncData(returnData, overCallBack)
    -- 保持本次登陆信息
    self:SaveSignInInfos(returnData)
    -- 同步全部存档
    UploadingGameDataHelper:UploadAllGameData()
    if overCallBack ~= nil then
        overCallBack()
    end
end

---@type fun():string
---@return string @ 后端返回的登录Token
function SaveDataHelper:GetLoginToken()
    return self._loginToken
end
