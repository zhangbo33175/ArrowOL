---@class ETimeCustomFormat @自定义时间格式
---@field Time_0m00s number @0m 00s 格式
---@field Time_0_00 number @0:00 格式
ETimeCustomFormat = {
    Time_0m00s = 1, --- 0m 00s
    Time_0_00 = 2 --- 0:00
}

---@class EFormatTimeString @时间格式化字符串
---@field Time_0d00h string @ 0d 00h 格式
---@field Time_0h00m string @ 0h 00m 格式
---@field Time_0m00s string @0m 00s 格式
---@field Time_0_00 string @0:00 格式
EFormatTimeString = {
    Time_0d00h = "%sd %02dh", --- 0d 00h
    Time_0h00m = "%sh %02dm", --- 0h 00m 
    Time_0m00s = "%sm %02ds", --- 0m 00s
    Time_0_00 = "%s:%02d", --- 0:00
}

---@class TimestampRecorder @一条记录
---@field timestamp number @ UTC时间戳
---@field callback function @ 回调

---@class GameTimeHelper :Singleton @时间工具
---@field TimestampRecords table @按先后顺序排序、带回调的时间戳集合
---@field WaittingExecute table @ 等待执行的任务
---@field EventCmdProcSecond string @注册事件用的Key
---@field TimestampRecordsCache table @缓存
---@field FutureTimeStamp number @ 未来时
---@field DayStr string @ 多语言文本 天
---@field HourStr string @ 多语言文本 小时
---@field MinStr string @ 多语言文本 分钟
---@field SecStr string @ 多语言文本 秒
local GameTimeHelper = class("GameTimeHelper", import("Singleton"))

---构造函数
---@type fun(args:table):void
---@param args table @自定义参数
function GameTimeHelper:ctor(args)
    GameTimeHelper.super.ctor(self, args)
    self.TimestampRecords = {}
    self.WaittingExecute = {}
    self.TimestampRecordsCache = {}
    self.EventCmdProcSecond = "EventCmd_ProcSecond"
    self.DayStr = nil
    self.HourStr = nil
    self.MinStr = nil
    self.SecStr = nil
    self:_InitLanguageText()

end

---创建函数
---@type fun(args:table):GameTimeHelper
---@param args table @自定义参数
---@return GameTimeHelper @时间工具
function GameTimeHelper:Create(args)
    local obj = GameTimeHelper.new(args)
    return obj
end

---初始化
---@type fun():void
function GameTimeHelper:Init()
    GameTimeHelper.super.Init(self)
    HonorEventHelper:AddListener(EventCmd.OnLanguageChange, handler(self, self._OnLanguageChange))
    HonorEventHelper:AddListener(self.EventCmdProcSecond, handler(self, self.ProcSecond))
    HonorTimerHelper:RegisterDeltaTime(self.EventCmdProcSecond, 1, "ProcedurePlaying", true)
end

---获取优先级
---@type fun():number
---@return number @优先级
function GameTimeHelper:Priority()
    return ESingletonPriority.GameTimeHelper
end

---清理数据
---@type fun():void
function GameTimeHelper:ClearData()
    GameTimeHelper.super.ClearData(self)

    self.TimestampRecords = {}
    self.WaittingExecute = {}
    self.TimestampRecordsCache = {}

    ---重设Future时间戳
    self:SetupFutureTimeStamp()
end

---多语言切换
function GameTimeHelper:_OnLanguageChange()
    self:_InitLanguageText()
end

---刷新多语言文本
function GameTimeHelper:_InitLanguageText()
    self.DayStr = GetLang('TimeShowDay')
    self.HourStr = GetLang('TimeShowHour')
    self.MinStr = GetLang('TimeShowMin')
    self.SecStr = GetLang('TimeShowSec')
end

---插入到检查列表
---@type fun(record :TimestampRecorder) :void
---@param record TimestampRecorder @ 记录
function GameTimeHelper:_Insert(record)
    local count = #self.TimestampRecords
    local min = 1
    local max = count
    while (count > 0 and max - min > 0)
    do
        local mid = min + math.floor((max - min) / 2)
        if self.TimestampRecords[mid].timestamp > record.timestamp then
            max = math.clamp(mid - 1, min, max)
        elseif self.TimestampRecords[mid].timestamp < record.timestamp then
            min = math.clamp(mid + 1, min, max)
        else
            min = mid
            break
        end
    end
    if self.TimestampRecords[min] == nil or self.TimestampRecords[min].timestamp >= record.timestamp then
        table.insert(self.TimestampRecords, min, record)
    else
        table.insert(self.TimestampRecords, min + 1, record)
    end
end

---注入一个定时任务
---@type fun(key : string ,timestamp:number,func :function):void
---@param key string @ 用于取消倒计时监听的key,可为空
---@param timestamp number @时间戳
---@param func function @时间到的回调
function GameTimeHelper:RegisterTime(key, timestamp, func)
    table.insert(self.TimestampRecordsCache, { key = key, timestamp = timestamp, callback = func })
end

---取消一个定时任务
---@type fun(key : string):void
---@param key string @ 用于取消的key
function GameTimeHelper:UnRegisterTime(key)
    --可能在缓存里
    for i = 1, #self.TimestampRecordsCache do
        local task = self.TimestampRecordsCache[i]
        if task.key and task.key == key then
            table.remove(self.TimestampRecordsCache, i)
            break
        end
    end
    --可能在记录里
    for i = 1, #self.TimestampRecords do
        local task = self.TimestampRecords[i]
        if task.key and task.key == key then
            table.remove(self.TimestampRecords, i)
            break
        end
    end
end

---秒周期函数
---@type fun():void
function GameTimeHelper:ProcSecond()
    ---拷贝进来
    local count = #self.TimestampRecordsCache
    for i = 1, count do
        self:_Insert(self.TimestampRecordsCache[i])
        self.TimestampRecordsCache[i] = nil
    end

    ---检查长度
    local total = #self.TimestampRecords
    if total == 0 then
        return
    end

    ---时间检查
    local idx = 0
    for i = 1, total do
        if self.TimestampRecords[i].timestamp > self:GetUTC0FutureTimeStamp() then
            break
        end
        idx = i
    end
    if idx == 0 then
        return
    end

    ---时间到，记录需要发送的回调
    ClearArray(self.WaittingExecute)
    for i = 1, idx do
        if self.TimestampRecords[i].key then
            PrintInfo(string.format("GameTimeHelper:ProcSecond : %s", self.TimestampRecords[i].key))
        end
        --self.TimestampRecords[i].callback()
        table.insert(self.WaittingExecute,self.TimestampRecords[i])
    end

    
    for i = 1, total do
        if i <= total - idx then
            self.TimestampRecords[i] = self.TimestampRecords[i + idx]
        else
            self.TimestampRecords[i] = nil
        end
    end

    ---发送回调事件
    for i = 1, #self.WaittingExecute do
        self.WaittingExecute[i].callback()
    end
    ClearArray(self.WaittingExecute)
end

---格式化时间
---@type fun(time:number,format:ETimeCustomFormat):string
---@param time number @时间
---@param format ETimeCustomFormat @格式
---@return string @格式化后的时间
function GameTimeHelper:FormatTime(time, format)
    if format == ETimeCustomFormat.Time_0m00s then
        local minute = math.floor(time / 60)
        local second = math.floor(time % 60)
        return string.format(EFormatTimeString.Time_0m00s, minute, second)
    elseif format == ETimeCustomFormat.Time_0_00 then
        local minute = math.floor(time / 60)
        local second = math.floor(time % 60)
        return string.format(EFormatTimeString.Time_0_00, minute, second)
    end
    return ""
end

---通用格式化
---@type fun(time:number,inTick:boolean):string
---@param time number @时间
---@param inTick boolean @ 处于倒计时中
---@return string @格式化后的时间
function GameTimeHelper:FormatTimeInCommonRule(time, inTick)
    if time >= 86400 then
        local day = math.floor(time / 86400)
        local hour = math.floor((time - day * 86400) / 3600)
        if hour == 0 then
            return string.format("%s%s", day, self.DayStr)
        else
            return string.format("%s%s %s%s", day, self.DayStr, hour, self.HourStr)
        end
    elseif time >= 3600 then
        local hour = math.floor(time / 3600)
        local minute = math.floor((time - hour * 3600) / 60)
        if minute == 0 then
            return string.format("%s%s", hour, self.HourStr)
        else
            return string.format("%s%s %s%s", hour, self.HourStr, minute, self.MinStr)
        end
    elseif time >= 60 then
        local minute = math.floor(time / 60)
        local second = math.floor(time % 60)
        if second == 0 and not inTick then
            return string.format("%s%s", minute, self.MinStr)
        else
            return string.format("%s%s %02d%s", minute, self.MinStr, second, self.SecStr)
        end
    else
        local second = math.floor(time % 60)
        return string.format("%02d%s", second, self.SecStr)
    end
end

--region 未来时
---登录游戏时设定一次，设定一个未来时间戳，记录玩家最未来的一次登录时间戳
function GameTimeHelper:SetupFutureTimeStamp()
    if not PbRootData.GameData.timestamps[ETimestampID.FutureTimeStamp] then
        ---初始化为一个本地时间
        PbRootData.GameData.timestamps[ETimestampID.FutureTimeStamp] = PbRootData.UserData.launchTime.current.utc0
    end

    ---游戏登录时间
    local loginTime = PbRootData.UserData.launchTime.current.utc0
    local futureTime = PbRootData.GameData.timestamps[ETimestampID.FutureTimeStamp]

    ---本次登录时间大于记录中的最未来的登录时间
    if futureTime < loginTime then
        ---用运行时间矫正登录时间（写入登录时间和本函数是在同一帧调用的）
        PbRootData.GameData.timestamps[ETimestampID.FutureTimeStamp] = loginTime - Unity.Time.realtimeSinceStartup
        PbIO.GameData.Save_timestamps()
    end

    ---写入到此
    self.FutureTimeStamp = PbRootData.GameData.timestamps[ETimestampID.FutureTimeStamp]
    PrintInfo(string.format("Future sub =%s", self:GetUTC0FutureTimeStamp() - TimerHelper:GetUTC0TimeStamp()))
    PrintInfo(string.format("Future format =  %s", TimerHelper:GetTimeStringByTimeStamp("yyyy-MM-dd HH:mm:ss", self.FutureTimeStamp)))
end

---退出游戏时、切到后台时将FutureTimeStamp刷新一次
function GameTimeHelper:RefreshFutureTimeStamp()
    if not self.FutureTimeStamp then
        ---防止初始化之前误刷
        return
    end
    PbRootData.GameData.timestamps[ETimestampID.FutureTimeStamp] = self:GetUTC0FutureTimeStamp()
    PbIO.GameData.Save_timestamps()
end

---返回一个未来时
---@type fun():number
---@return number @ 未来时
function GameTimeHelper:GetUTC0FutureTimeStamp()
    if not self.FutureTimeStamp then
        PrintError("GameTimeHelper => SetupFutureTimeStamp must be call...")
        self.FutureTimeStamp = PbRootData.UserData.launchTime.current.utc0
    end

    return self.FutureTimeStamp + Unity.Time.realtimeSinceStartup
end
--endregion

return GameTimeHelper
