---注册挂起/恢复应用时回调
---@type fun(pause:boolean):void
---@param pause boolean @ 应用是否挂起
function ApplicationPauseCallback(pause)
    -- 挂起应用时的处理
    if pause == true then
        PrintInfo("回调 Lua 应用挂起")
    else  -- 恢复应用时的处理
        PrintInfo("回调 Lua 应用恢复")
        DOTween.Sequence():AppendInterval(0.5):AppendCallback(function()
            HonorTimerHelper:SetTimeStamp(nil) -- 设置本地时间戳
        end)
    end
    GameApplicationPauseCallback(pause)
end

---退出应用时回调（ApplicationQuit发生在所有GameObject销毁之前，该Unity的C#接口在个别安卓设备下可能无效）
---@type fun():void
function ApplicationQuitCallback()
    PrintInfo("回调 Lua 应用退出")
    GameApplicationQuitCallback()
end

---原生按键抬起时回调
---@type fun(keyCode:any):void
---@param keyCode any @UnityEngine.KeyCode按键类型枚举
function KeyboardsUpCallback(keyCode)
    PrintInfo("回调 Lua 原生按键抬起：keyCode = "..keyCode:GetHashCode())
    -- 目前暂时只支持Escape按键回调
    GameKeyboardsUpCallback(keyCode)
end

---接收C#层事件回调相关处理
---@type fun(eventArgs: Honor.Runtime.EventParams):void
---@param eventArgs Honor.Runtime.EventParams @C#层事件参数
function ReceiveCsEventCallback(eventArgs)
    -- 监听屏幕方向变动事件
    if eventArgs.Cmd == Honor.EventCmd.ScreenOrientationChanged then
        PrintInfo("ScreenOrientationChanged  LastOrientation = " .. eventArgs:GetInt("LastOrientation") ..
                "  CurOrientation = " .. eventArgs:GetInt("CurOrientation") ..
                "  BangsWidth = " .. eventArgs:GetString("BangsWidth") ..
                "  BangsHeight = " .. eventArgs:GetString("BangsHeight"))
        HonorEventHelper:Fire(FwEventCmd.ScreenOrientationChanged, eventArgs)
    end
    GameReceiveCsToEventCallback(eventArgs)
end
