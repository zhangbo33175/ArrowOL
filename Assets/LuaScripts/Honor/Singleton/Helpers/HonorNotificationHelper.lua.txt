---@class NotificationHelper @通知助手
local HonorNotificationHelper = class("HonorNotificationHelper")

---构造函数
---@type fun(args:table):void
---@param args table @自定义参数
function HonorNotificationHelper:ctor(args)
	-- 发送通知x条后强制推送下消息
	self._nativeNotificationMessageMaxCount = 5
	-- 每x秒检测一次本地的消息缓存通知消息
	self._nativeNotificationMessageTimeDelta = 30
	-- 当前通知消息的次数
	self._nativeNotificationMessageCurCount = 0
	self:Init()
end

---创建函数
---@type fun(args:table):NotificationHelper
---@param args table @自定义参数
---@return NotificationHelper @存档助手实例
function HonorNotificationHelper:Create(args)
	local obj = HonorNotificationHelper.new(args)
	return obj
end

---初始化
---@type fun(args:table):void
---@param args table @自定义参数
function HonorNotificationHelper:Init(args)
	-- 监听通知
	HonorEventHelper:AddListener(EventCmd.NotificationMessage, handler(self, self.RequestNotificationMessage))
	-- 发送轮询发送消息的监听
	HonorTimerHelper:RegisterDeltaTime(EventCmd.NotificationMessage, self._nativeNotificationMessageTimeDelta, "ProcedurePlaying", true)

end

---获取通知功能是否开启
---@type fun():boolean
---@return boolean @返回是否开启了通知功能，SystemConfig表控制
function HonorNotificationHelper:CheckNotifyFuncIsOpen()
	if Unity.Application.platform == Unity.RuntimePlatform.Android then
		return SystemConfigHelper:GetSystemConfig("ANDROID_NOTIFY_FUNCTION_OPEN")
	elseif Unity.Application.platform == Unity.RuntimePlatform.IPhonePlayer then
		return SystemConfigHelper:GetSystemConfig("IOS_NOTIFY_FUNCTION_OPEN")
	else
		return SystemConfigHelper:GetSystemConfig("ANDROID_NOTIFY_FUNCTION_OPEN")
	end

end
