---Lua热重载
---@type fun(filename:string):void
---@param filename string @Lua文件名
function __RELOAD_LUA_HOTFIX__(filename)
    PrintDebug("开始热重载Lua脚本: {0}", filename)
    local oldModule
    if package.loaded[filename] then
        oldModule = package.loaded[filename]
        package.loaded[filename] = nil
    else
        PrintDebug("该Lua脚本从未加载过: {0}",filename)
        return
    end
    local ok, err = pcall(require, filename)
    if not ok then
        package.loaded[filename] = oldModule
        PrintDebug("热重载Lua脚本失败，错误信息：", err)
        return
    end

    local newModule = package.loaded[filename]
    
    local updated_tables = {}
    __RELOAD_LUA_HOTFIX_UPDATE_TABLE__(newModule, oldModule, updated_tables)

    if "table" == type(oldModule) and oldModule.OnReload ~= nil then
        oldModule:OnReload()
    end

    PrintDebug("Lua脚本替换成功！")
    package.loaded[filename] = oldModule
end

---更新Lua方法
---@type fun(new_func:function, old_func:function):void
---@param new_func function @新Lua方法
---@param old_func function @旧Lua方法
function __RELOAD_LUA_HOTFIX_UPDATE_FUNC__(new_func, old_func)
    if "function" == type(new_func) and "function" == type(old_func) then
        -- 获取旧方法的upvalues（upvalue：就是一个函数使用了不在这个函数内的变量）
        local old_upvalue_map = {}
        for i = 1, math.huge do
            local name, value = debug.getupvalue(old_func, i)
            if not name then break end
            old_upvalue_map[name] = value
        end
        -- 用旧的upvalues来更新新的upvalues（upvalue：就是一个函数使用了不在这个函数内的变量）
        for i = 1, math.huge do
            local name, value = debug.getupvalue(new_func, i)
            if not name then break end
            PrintDebug("设置upvalue:{0}", name)
            local old_value = old_upvalue_map[name]
            if old_value then
                debug.setupvalue(new_func, i, old_value)
            end
        end
    end
end

---更新Lua的table
---@type fun(new_table:table, old_table:table, updated_tables:table):void
---@param new_table table @新table
---@param old_table table @旧table
---@param updated_tables table @已更新的table
function __RELOAD_LUA_HOTFIX_UPDATE_TABLE__(new_table, old_table, updated_tables)
    if "table" == type(new_table) and "table" == type(old_table) then
        -- 比较新旧table，并更新旧table
        for key, value in pairs(new_table) do
            local old_value = old_table[key]
            local type_value = type(value)
            if type_value == "function" then
                __RELOAD_LUA_HOTFIX_UPDATE_FUNC__(value, old_value)
                old_table[key] = value
            elseif type_value == "table" then
                if ( updated_tables[value] == nil ) then
                    updated_tables[value] = true
                    __RELOAD_LUA_HOTFIX_UPDATE_TABLE__(value, old_value, updated_tables)
                end
            end
        end
        -- 更新元表
        local old_meta = debug.getmetatable(old_table)
        local new_meta = debug.getmetatable(new_table)
        if type(old_meta) == "table" and type(new_meta) == "table" then
            __RELOAD_LUA_HOTFIX_UPDATE_TABLE__(new_meta, old_meta, updated_tables)
        end
    end
end
